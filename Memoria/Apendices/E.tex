\chapter{Corpus de frases simples}
\label{ap1:E}
\begin{enumerate}


 \item Él bebe agua.
 \item Yo soy morena. 
 \item Ellos parecen contentos. 
 \item María compra galletas. 
 \item Yo bebo agua. 
 \item Yo como carne.
 \item Ellos compraron aceite. 
 \item Él venderá refrescos
 \item Ella ganó la carrera. 
 \item Yo haré la cama.
 \item Mi tía y mi hermana beberán agua.
 \item Ella y él comieron arroz. 
 \item María y Javier nadan.
 \item Mi hermana y mi tía están de vacaciones en la playa.
 \item Los niños comerán chocolate negro.
 \item Las niñas jugarán en el parque. 
 \item Mis tíos irán al supermercado en coche.
 \item Los niños jugarán con la pelota. 
 \item Los ganadores recibirán un trofeo de oro. 
 \item Él pasa ante la casa todos los días.
 \item Él pasea ante la casa por la mañana.
 \item Mañana yo iré al médico.
 \item Él comió mucho en la fiesta. 
 \item Él se comió la hamburguesa aquí. 
 \item Mi tía conduce muy despacio. 


\end{enumerate}
